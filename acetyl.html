def find_CC(a):
    data = pd.DataFrame()
    for seq in a:

        for j, find in enumerate(regex.finditer('C-*[^-]-*[^-]-*C', str(seq.seq), overlapped=True)):
                data.at[i, f'start{j}'] = find.span()[0]
                data.at[i,   f'seq{j}'] = find.group()
    for seq in a:

        for j, find in enumerate(regex.finditer('C-*[^-]-*[^-]-*C', str(seq.seq), overlapped=True)):

    return data

    #  inct

    group.inclusion = final_incl
    return
